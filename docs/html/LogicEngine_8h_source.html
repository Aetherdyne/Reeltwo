972  //function to calculate all colors based on the chosen colorPalNum
973  //this will get called during setup(), and also if we've remotely commanded to change color palettes
974  //NEW: added a "bright" parameter, to allow front and rear brightness to be adjusted independantly
975  void calculateAllColors(byte colorPalNum, byte brightVal)
976  {
977  //we define both all palettes, in case the user wants to try out rear colors on the front etc
978  // note that these are not RGB colors, they're HSV
979  // for help calculating HSV color values see http://joymonkey.com/logic/
980  static const HSVColor keyColors[PAL_COUNT][4] PROGMEM = {
981  { {170,255,0} , {170,255,136} , {170,255,255} , {170,0,255} }, //front colors
982  { {87,206,0} , {87,206,105} , {45,255,184} , {0,255,250} }, //rear colors (hues: 87=bright green, 79=yellow green, 45=orangey yellow, 0=red)
983  { {0,255,0} , {0,255,85} , {0,255,170} , {0,255,255} }, //monotone (black to solid red)
984  { {0,255,0} , {0,255,250} , {40,255,0} , {40,255,250} } //dual color red and yellow
985  };
986  //take a set of 4 key colors from keyColors[3][4][3] and generate 16 colors
987  // 328P will only have one set of full colors, Teensy will have two
988  for (byte kcol = 0; kcol < PAL_COUNT; kcol++)
989  {
990  //go through each Key color
991  const HSVColor* wc = &keyColors[colorPalNum][kcol];
992  HSVColor workColor = {
993  pgm_read_byte(&wc->h),
994  pgm_read_byte(&wc->s),
995  pgm_read_byte(&wc->v)
996  };
997  byte tnum = kcol + fTweens * kcol;
998  HSVColor* tc = &fAllColors[tnum];
999  tc->h = workColor.h;
1000  tc->s = workColor.s;
1001  //Value (V) is adjusted down to whatever brightness setting we've specified
1002  brightVal = min(brightVal, MAX_BRIGHTNESS);
1003  tc->v = map8(workColor.v, MIN_BRIGHTNESS, brightVal);
1004  if (tnum + 1 != fTotalColors)
1005  {
1006  const HSVColor* hsvptr = &keyColors[colorPalNum][kcol + 1];
1007  for (byte el = 0; el < 3; el++)
1008  {
1009  //loop through H, S and V from this key to the next
1010  int perStep = int(pgm_read_byte(&hsvptr->c[el]) - workColor.c[el]) / (fTweens + 1);
1011  if (perStep != 0)
1012  {
1013  for (byte tweenCount = 1; tweenCount <= fTweens; tweenCount++)
1014  {
1015  byte val = workColor.c[el] + tweenCount * perStep;
1016  tc[tweenCount].c[el] = (el == 2) ?
1017  map8(val, MIN_BRIGHTNESS, brightVal) : val;
1018  }
1019  }
1020  else
1021  {
1022  //tweens for this element (h,s or v) don't change between this key and the next, fill em up
1023  for (byte tweenCount = 1; tweenCount <= fTweens; tweenCount++)
1024  {
1025  tc[tweenCount].c[el] = (el == 2) ?
1026  map8(workColor.c[el], MIN_BRIGHTNESS, brightVal) : workColor.c[el];
1027  }
1028  }
1029  }
1030  }
1031  }
1032  }
