339  servoSetMask |= (uint32_t)pgm_read_byte(&step->servo9_16) << 16;

342  uint32_t ms = pgm_read_word(&fSequence[fIndex].cs) * 10L;
343  fNextStepMS = currentTime + ms;
344  fDispatch.moveServoSetTo(fServoGroupMask, servoSetMask, 0, fSpeedMinMS, fSpeedMaxMS, fMaxPulse, fMinPulse);
