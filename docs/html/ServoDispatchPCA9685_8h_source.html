400  {
401  if (finishTime != 0)
402  {
403  if (timeNow < startTime)
404  {
405  /* wait */
406  }
407  else if (timeNow >= finishTime)
408  {
409  posNow = finishPos;
410  doMove(dispatch, timeNow);
411  init();
412  }
413  else if (lastMoveTime != timeNow)
414  {
415  uint32_t timeSinceLastMove = timeNow - lastMoveTime;
416  uint32_t denominator = finishTime - lastMoveTime;
417  float fractionChange = float(timeSinceLastMove)/float(denominator);
418 
419  int distanceToGo = finishPos - posNow;
420  float distanceToMove = float(distanceToGo) * fractionChange;
421  int distanceToMoveInt = int(distanceToMove);
422 
423  if (abs(distanceToMoveInt) > 1)
424  {
425  posNow = posNow + distanceToMoveInt;
426  doMove(dispatch, timeNow);
427  }
428  }
